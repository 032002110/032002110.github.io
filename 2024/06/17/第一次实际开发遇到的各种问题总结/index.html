enum AddVipErrorCode:String {
case Error_UserDevice ///“领取失败，当前账号与设备已解除绑定”
case AddVipClose ///“暂不支持领取”，提示并隐藏领取入口
case Login ///提示并自动退出当前账号
case Error_AnotherAccount ///“领取失败，权益已被其他账号领取”，提示并移除该设备的VIP权益领取入口
}

@objc protocol BBPenAddVipContentViewDelegate:NSObjectProtocol {
////确认领取
@objc optional func addvipContentViewDidConfirmReceive(deviceID:String,addVipID:Int)
}

class BBPenAddVipContentView: UIView {

weak var popup:BBPenPopup?
weak var delegate:BBPenAddVipContentViewDelegate?
var deviceID:String = ""
var addVipID:Int = 0
convenience init(deviceID:String,addVipID:Int) {
    self.init()
    self.deviceID = deviceID
    self.addVipID = addVipID
}

lazy var backgroundImgView: UIImageView = {
    let imageView = UIImageView(image: UIImage.bb_bundle_imageName(name: "pen_addvip_content_bg"))
    return imageView
}()

lazy var foregroundImgView:UIImageView = UIImageView(image: .bb_bundle_imageName(name: "pen_addvip_confirm_fore"))

lazy var titleLabe:UILabel = UILabel().then { label in
    let text = "宝宝巴士SVIP领取须知"
    let prefix = "宝宝巴士SVIP"
    let attributedText = NSMutableAttributedString(string: text,attributes: [.foregroundColor:"#5E3823".bbColor,.font:UIFont.systemFont(ofSize: IS_IPAD ? 22.bbScale : 18.bbScale, weight: .semibold)])
    attributedText.addAttributes([.foregroundColor:"#DB6B00".bbColor], range: NSRange(location: 0, length: prefix.count))
    label.attributedText = attributedText
}

lazy var useContentView:BBPenAddVipContentSubView = BBPenAddVipContentSubView().then { contentView in
    contentView.titleLabel.text = "SVIP权益仅限绑定设备的账号使用"
    let content = "请确认是否由当前账号领取权益，若需要其他账号领取，需其他账号添加设备后在APP内操作；权益一经领取，仅领取的账号享有SVIP权益，无法通过更换手机号等方式进行权益转让。"
    let paraStyle = NSMutableParagraphStyle()
    paraStyle.lineSpacing = IS_IPAD ? 4.bbScale : 3.bbScale
    let attr = NSMutableAttributedString(string: content,attributes: [.foregroundColor:UIColor.cccGray2,.font:UIFont.systemFont(ofSize: IS_IPAD ? 15.bbScale : 14.bbScale, weight: .regular),.paragraphStyle:paraStyle])
    contentView.contentLabel.attributedText = attr
    contentView.layer.cornerRadius = 14.bbScale
    contentView.layer.masksToBounds = true
}

lazy var withDrawContentView:BBPenAddVipContentSubView = BBPenAddVipContentSubView().then { contentView in
    contentView.titleLabel.text = "SVIP权益回收"
    let content = "SVIP权益仅限购买并绑定设备的用户享有，若领取后发生退货情况，宝宝巴士有权回收已发放的权益。"
    let paraStyle = NSMutableParagraphStyle()
    paraStyle.lineSpacing = IS_IPAD ? 4.bbScale : 3.bbScale
    let attr = NSMutableAttributedString(string: content,attributes: [.foregroundColor:UIColor.cccGray2,.font:UIFont.systemFont(ofSize: IS_IPAD ? 15.bbScale : 14.bbScale, weight: .regular),.paragraphStyle:paraStyle])
    contentView.contentLabel.attributedText = attr
    contentView.layer.cornerRadius = 14.bbScale

}

    button.setImage(.bb_bundle_imageName(name: "pen_addvip_selected"), for: [.selected,.highlighted])
    button.setTitleColor(.black, for: .normal)
    button.setTitleColor(.black, for: .highlighted)
    button.titleLabel?.font = .systemFont(ofSize: IS_IPAD ? 16.bbScale : 14.bbScale, weight: .regular)
    button.setImagePosition(.left, spacing: 4.bbScale)
    button.imageRect = CGRect(x: 0, y: 0, width: width, height: width)

@objc func agreeAction(_ button:BBSButton) {
    button.isSelected = !button.isSelected
}

@objc func cancelAction() {

// if IS_IPAD {
// self.viewController().dismiss(animated: true)
// } else {
// self.popup?.dismiss()
// }
self.popup?.dismiss()
}

@objc func confirmAction() {
    if agreeButton.isSelected == false {
        self.hideToast(false)
        self.toastMessage("请先阅读须知")
        return
    }

// if IS_IPAD {
//// self.viewController().dismiss(animated: true)
// } else {
// popup?.dismiss()
// }
popup?.dismiss()
delegate?.addvipContentViewDidConfirmReceive?(deviceID: deviceID, addVipID: addVipID)
}

override init(frame: CGRect) {
    super.init(frame: frame)
    self.setupViews()
}

required init?(coder: NSCoder) {
    super.init(coder: coder)
    self.setupViews()
}

func setupViews() {
    
    let leftInset = IS_IPAD ? 40.bbScale : 20.bbScale
    
    addSubview(backgroundImgView)
    backgroundImgView.snp.makeConstraints { make in
        make.edges.equalToSuperview()
        make.width.equalTo(IS_IPAD ? SCREEN_MIN : 375.bbScale)
        make.height.equalTo(IS_IPAD ? 451.gsScale : 511.bbScale)
    }
    
    if IS_IPAD {
        addSubview(foregroundImgView)
        foregroundImgView.snp.makeConstraints { make in
            make.left.right.equalToSuperview()
            make.top.equalToSuperview().offset(-20.bbScale)
            make.height.equalTo(165.bbScale)
        }
    }
    
    addSubview(titleLabe)
    titleLabe.snp.makeConstraints { make in
        make.top.equalTo(IS_IPAD ? 73.bbScale : 44.bbScale)
        make.left.equalTo(leftInset)
        make.height.equalTo(IS_IPAD ? 31.bbScale : 25.bbScale)
    }
    
    addSubview(useContentView)
    useContentView.snp.makeConstraints { make in
        make.top.equalTo(titleLabe.snp.bottom).offset(IS_IPAD ? 40.bbScale : 16.bbScale)
        make.left.equalTo(leftInset)
        make.right.equalTo(-leftInset)
    }
    
    addSubview(withDrawContentView)
    withDrawContentView.snp.makeConstraints { make in
        make.top.equalTo(useContentView.snp.bottom).offset(12.bbScale)
        make.left.right.equalTo(useContentView)
    }
    
    addSubview(agreeButton)
    agreeButton.addTarget(self, action: #selector(agreeAction(_:)), for: .touchUpInside)
    agreeButton.snp.makeConstraints { make in
        make.left.equalTo(leftInset)
        if kIsIPadX {
            make.top.equalTo(withDrawContentView.snp.bottom).offset(IS_IPAD ? 100.bbScale : 16.bbScale)
        }else {
            make.top.equalTo(withDrawContentView.snp.bottom).offset(IS_IPAD ? 153.bbScale : 16.bbScale)
        }
       
        make.height.equalTo(agreeButton.height)
    }
    
    addSubview(cancelButton)
    cancelButton.addTarget(self, action: #selector(cancelAction), for: .touchUpInside)
    cancelButton.snp.makeConstraints { make in
        make.top.equalTo(agreeButton.snp.bottom).offset(IS_IPAD ? 12.bbScale : 20.bbScale)
        make.left.equalTo(leftInset)
        make.width.equalTo(IS_IPAD ? 130.gsScale : 120.bbScale)
        make.height.equalTo(IS_IPAD ? 50.gsScale : 44.bbScale)
    }
    
    addSubview(confirmButton)
    confirmButton.addTarget(self, action: #selector(confirmAction), for: .touchUpInside)
    confirmButton.snp.makeConstraints { make in
        make.left.equalTo(cancelButton.snp.right).offset(IS_IPAD ? 24.bbScale : 16.bbScale)
        make.centerY.equalTo(cancelButton)
        make.right.equalTo(-leftInset)
        make.height.equalTo(IS_IPAD ? 50.gsScale : 44.bbScale)
    }
}

deinit {
    debugPrint("AddVipContentViewDeinit")
}

}

lazy var titleLabel:UILabel = UILabel().then { label in
    label.textColor = .cccGray1
    label.font = .systemFont(ofSize: IS_IPAD ? 18.bbScale : 15.bbScale, weight: .semibold)
}

lazy var contentLabel:UILabel = UILabel().then { label in
    label.textColor = .cccGray2
    label.numberOfLines = 0
    label.font = .systemFont(ofSize: IS_IPAD ? 15.bbScale : 14.bbScale, weight: .regular)
}

override init(frame: CGRect) {
    super.init(frame: frame)
    self.setupViews()
}

required init?(coder: NSCoder) {
    super.init(coder: coder)
    self.setupViews()
}

func setupViews() {
    
    backgroundColor = .white
    
    addSubview(titleLabel)
    titleLabel.snp.makeConstraints { make in
        make.left.top.equalTo(16.bbScale)
        make.right.equalTo(-16.bbScale)
        make.height.equalTo(IS_IPAD ? 25.bbScale : 21.bbScale)
    }
    
    addSubview(contentLabel)
    contentLabel.snp.makeConstraints { make in
        make.top.equalTo(titleLabel.snp.bottom).offset(12.bbScale)
        make.left.equalTo(16.bbScale)
        make.right.bottom.equalTo(-16.bbScale)
    }
}

}
