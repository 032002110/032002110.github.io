音频模块 

// 检查网络连接状态，如果没有网络连接，则显示网络错误提示并返回
if BBSAppManager.default().status == .notReachable {
    netErrorToast()
    return
}
// 检查登录状态
if BBSAppManager.default().isLogin() == false {
    BBSRouterManager.openLoginVC("收藏音频") // 打开登录界面
    return // 如果未登录，直接返回
}

// 执行收藏操作
BBSAudioManager.default().collectMedia(!button.isSelected) { [weak self] isCollect, error in
    // 如果收藏操作失败
    if error != nil {
        GSCommonHelper.showToast("操作失败，请检查网络") // 显示操作失败提示
        self?.resetCollectAnimation() // 重置收藏动画
        self?.collectMediaButtons(!isCollect) // 更新按钮状态
        return // 返回
    }
    
    // 如果未收藏，直接返回
    if !isCollect {
        return
    }
    
    // 如果配置允许显示评价视图，显示评价视图
    if BBSAppManager.default().configuration.isCanShowEvaluateView {
        BBSEvaluateViewController.showPictrueBookPlayeEvaluateView() // 显示评价视图
    }
}
